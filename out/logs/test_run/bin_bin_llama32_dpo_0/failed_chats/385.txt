def priority(item: float, bins: np.ndarray) -> np.ndarray:

    """
    # Calculate the minimum capacity remaining in all bins
    min_cap_remaining = np.minimum(..., bins - item)

    # If all bins are empty, return 0
    else:
        return np.zeros_like(bins)
