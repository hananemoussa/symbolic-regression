```python
import numpy as np

def priority(item: float, bins: np.ndarray, threshold: float = 1.0) -> np.ndarray:
    """
    Returns the priority score for each bin based on the item size and the bin capacity.

    Args:
        item: Size of the item to be added to the bin.
        bins: Array of capacities for each bin.
        threshold: Default value for bin size threshold. Default is 1.0.

    Returns:
        Array of same size as bins with the priority score of each bin.
    """
    # Calculate the bin size as the threshold plus 1
    bin_size = threshold + 1
    
    # Compute the priority score for each bin
    score = (bins - item) / bin_size
    
    # Use absolute value to ensure a positive score
    score = np.abs(score)
    
    return score
```
