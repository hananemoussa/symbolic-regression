```python
import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.
    
    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.
    
    Return:
        Array of same size as bins with cumulative sums representing optimal number of items that can fit in each bin.
    """
    # Calculate cumulative sums
    cumulative_sums = bins.cumsum()
    
    # Use cumulative sums to calculate priority
    priority = -(cumulative_sums - item)
    
    return priority
```
