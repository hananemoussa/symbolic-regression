import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Calculate the minimum required capacity for each bin
    required_capacity = np.ceil(item / np.min(bins))

    # Calculate the remaining capacity for each bin after adding the item
    remaining_capacity = np.array(bins) - required_capacity

    # Calculate the priority score for each bin by the available capacity
    return remaining_capacity

Then, we calculate the remaining capacity for each bin after adding the item, by subtracting the required capacity from the minimum required capacity.
