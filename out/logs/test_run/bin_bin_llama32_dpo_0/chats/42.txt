import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Sort the items by their size in descending order
    sorted_items = np.argsort(-bins)

    # Initialize the priority scores with negative infinity
    priority_scores = np.zeros_like(bins)

    # Assign priority scores to items based on their relative size
    for i in sorted_items:
        # If the item fits into the bin, assign a high priority score
        if bins[i] >= item:
            priority_scores[i] = 1.0
        # Otherwise, assign a lower priority score
        else:
            priority_scores[i] = -np.inf

    return priority_scores
