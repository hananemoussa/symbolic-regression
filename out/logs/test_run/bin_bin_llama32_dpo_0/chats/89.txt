import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with cumulative sum of bins as priority score.
    """
    bins = bins.astype(int)  # Ensure bins are integers for cumulative sum
    cumulative_sums = bins.cumsum()  # Calculate cumulative sum of bins

    return cumulative_sums - item  # Assign score to each bin
