To create an improved version of the `priority()` function, we will leverage a different approach to calculate the priority. import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:

    """
    # Calculate the fractional value of each bin
    bin_frac = bins / bins.sum()

    # Create an array of priorities where item size is subtracted for small fractional values
