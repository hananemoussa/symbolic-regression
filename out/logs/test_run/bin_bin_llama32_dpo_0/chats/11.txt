```python
import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Calculate the remaining capacity in the bin
    remaining_capacity = bins - item
    
    # Use the negative of the remaining capacity as the priority weight
    priority_weights = -remaining_capacity
    
    # Sort the priority weights in descending order and assign them to the bins
    sorted_indices = np.argsort(priority_weights)[::-1]
    bins_sorted = bins[sorted_indices]
    priority_weights_sorted = priority_weights[sorted_indices]
    
    return priority_weights_sorted
```

1.  Calculating the remaining capacity in the bin.
2.  Using the negative of the remaining capacity as the priority weight.
3.  Sorting the priority weights in descending order and assigning them to the bins.
