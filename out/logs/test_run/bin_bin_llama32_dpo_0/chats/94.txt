The idea behind this function is to allocate items to bins in a way that maximizes the utilization of bins, given their capacities. The key insight here is to place items in bins where their priority score is highest, as items with higher priority scores should be allocated to bins with smaller capacities.

import numpy as np
import random

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    # Calculate the priority score of each bin
    bin_priority = -(bins - item)
    
    # Sort bins by their priority scores in descending order
    sorted_bins = np.argsort(bin_priority)[::-1]
    
    # Allocate items to bins
    allocation = np.zeros_like(bins, dtype=int)
    item_distribution = np.zeros(len(bins), dtype=int)
    
    # Assign items to bins in the sorted order
    for i, bin_index in enumerate(sorted_bins):
        allocation[bin_index] = item_distribution[i] + 1  # Add 1 to account for the last bin
        item_distribution[i] += 1
    
    return allocation

# Example usage
