def priority(item: float, bins: np.ndarray) -> np.ndarray:

    Returns a priority with which the item should be added to each bin.

    This function uses a different approach than the existing functions. It calculates the priority by taking the negative of the bin index. This is because adding an item to the first bin should have a lower priority than adding it to the second bin, and so on.

    Returns:
        np.ndarray: A numpy array of priority scores for each bin.
    """

    return priority_scores
