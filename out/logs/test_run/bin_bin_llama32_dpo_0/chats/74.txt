import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns a priority score for each bin, indicating how much of the item should be added to it.

    Args:
        item (float): Size of the item to be added.
        bins (np.ndarray): Array of capacities for each bin.

    Return:
        np.ndarray: Array of priority scores for each bin.
    """
    # Calculate the priority score by adding the item to each bin's capacity
    # and subtracting the bin's capacity from the total capacity of all bins
    # This ensures that the item is distributed efficiently across all bins
    priority_scores = np.where(bins > item, item + bins - item, bins - item)
    
    return priority_scores

# Example usage

print(f"Priority scores for each bin: {score}")

def priority(item: float, bins: np.ndarray, weights: np.ndarray) -> np.ndarray:
