import numpy as np

def binary_search_array(arr, target):
    """
    Returns the index where the target element should be inserted in the array to maintain sorted order.
    """
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = left + (right - left) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return left

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Sort bins in ascending order based on their capacities
    bins = np.sort(bins)

    # Use binary search to find the bin with the maximum capacity
    max_idx = binary_search_array(bins, np.max(bins))
    max_capacity = bins[max_idx]
    score = max_idx * max_capacity - item

    # Calculate the score for each bin
    scores = np.ones_like(bins) * score
    for i, capacity in enumerate(bins):
        if capacity > 0:
            scores[i] -= (i + 1) * capacity
    return scores
