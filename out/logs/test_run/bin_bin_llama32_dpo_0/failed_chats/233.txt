```python
import numpy as np
from typing import np

def greedy_priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Greedy function that selects the bin with the smallest remaining capacity

    Args:
        item: Size of the item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
```

We can further optimize this function by using a binary search approach. This involves finding the bin with the smallest remaining capacity in a single iteration.

```python
def binary_search_priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Binary search function that selects the bin with the smallest remaining capacity
    when adding the given item.

    Args:
        item: Size of the item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with the index of the bin with the smallest remaining capacity.
    """
    low, high = 0, bins.shape[0] - 1
    while low < high:
        mid = (low + high) // 2
        if bins[mid] - item > bins[mid + 1] - item:
            low = mid + 1
        else:
            high = mid
    return low
```
