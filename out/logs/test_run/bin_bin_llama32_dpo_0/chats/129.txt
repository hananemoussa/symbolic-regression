import numpy as np

def priority(item: float, bins: np.ndarray, growth_rate: float = 0.5) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.
        growth_rate: Growth rate of the bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Initialize an empty array to store the priority scores
    priority_scores = np.zeros_like(bins)

    # The default growth rate is 0.5, so we set the initial growth rate to this value
    initial_growth_rate = growth_rate

    # Initialize the bin growth rate to the default growth rate
    current_growth_rate = initial_growth_rate

    # Iterate over each bin
    for i, bin_capacity in enumerate(bins):
        # Calculate the new bin capacity after adding the item
        new_capacity = bin_capacity + item

        # Calculate the priority score for the current bin
        priority_score = -np.log(np.exp(current_growth_rate) - np.log(np.exp(current_growth_rate - new_capacity)))

        # Update the current bin's growth rate
        current_growth_rate = growth_rate * current_growth_rate

        # Append the new bin's priority score to the array
        priority_scores[i] = priority_score

    return priority_scores
