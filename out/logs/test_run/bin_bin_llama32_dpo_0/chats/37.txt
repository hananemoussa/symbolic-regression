import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Calculate the expected number of items in each bin
    # if we distribute the item evenly
    expected_items = (bins + 1) / 2
    
    # Calculate the residual items that are not in bins
    residual_items = item - expected_items
    
    # Calculate the priority score as the residual items, relative to the expected items
    priority_score = residual_items / expected_items
    
    # Clip the priority score to ensure it falls within the valid range
    # (i.e., not negative)
    priority_score = np.clip(priority_score, 0, 1)
    
    return priority_score
