```python
import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns a list of scores representing the priority of each bin with which the item should be added.

    Args:
    item: The size of the item to be added to the bin.
    bins: The array of capacities for each bin.

    Return:
    A list of scores representing the priority of each bin.
    """
    # Calculate the minimum capacity of bins left
    left_bins = np.concatenate((np.ones(len(bins)), np.zeros(len(bins) - len(bins)), bins))
    
    # Calculate the rank of each bin based on the number of items it can hold
    ranks = np.argsort(left_bins)
    
    # Create the scores array by subtracting the item size from the rank of each bin
    scores = ranks * -item
    
    return scores
```
