import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    This function calculates the priority with which an item should be added to each bin.
    
    Args:
        item (float): The size of the item to be added to the bin.
        bins (np.ndarray): The array of capacities for each bin.

    Returns:
        np.ndarray: An array of same size as bins with priority score of each bin.
    """
    
    # Calculate the relative priority of each bin
    # This is done by subtracting the capacity of each bin from the item size, 
    # and then negating it, effectively giving a negative priority to the bin with the largest gap
    priority = -(bins - item)
    
    return priority
