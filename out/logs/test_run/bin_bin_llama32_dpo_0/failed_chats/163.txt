            dp[i, j] = np.inf

            for prev_bin in range(i):

                prev_wait_time = dp[prev_bin, j - bins[prev_bin]]

                dp[i, j] = min(dp[i, j], dp[prev_bin, j - bins[prev_bin]] + bins[prev_bin] - item)
