import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns the optimal priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with optimal priority for each bin.
    """
    # Sort items by their probabilities (as a way to ensure optimal distribution)
    probabilities = np.sort(bins) / bins.sum()

    # Calculate the optimal priority
    # We use the -log(probability) formula to ensure that we add the item to the bin with the highest probability
    return -np.log(probabilities)
