```python
def priority_greedy(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Sort bins in ascending order of their capacities
    sorted_bins = np.argsort(bins)
    
    # Calculate the ratio of item size to bin capacity
    ratios = item / bins[sorted_bins]
    
    # Use the greedy algorithm to assign items to bins
    priority_scores = np.zeros_like(ratios)
    
    for i, ratio in enumerate(ratios):
        priority_scores[i] = 1 / ratio
    
    return priority_scores
```
