```python
import numpy as np

def priority(item: float, bins: np.ndarray, capacity_decrease_factor=0.5, decrease_percentage=20) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.
        capacity_decrease_factor: Factor by which the capacity decreases for each bin.
        decrease_percentage: Percentage of capacity decrease.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Calculate the difference in capacity between the item and the maximum capacity
    capacity_diff = item - np.max(bins)
    
    # Calculate the percentage decrease in capacity
    percentage_decrease = 100 - int(capacity_diff * decrease_percentage / 100)
    
    # Calculate the priority score as the negative of the percentage decrease
    priority = -(percentage_decrease / 100)
    
    # Map the priority score to a valid array of capacities
    bins[priority < 0] = bins[priority < 0] - capacity_decrease_factor * bins[priority < 0]
    bins[priority >= 0] = bins[priority >= 0] + capacity_decrease_factor * bins[priority >= 0]
    
    return bins
```
