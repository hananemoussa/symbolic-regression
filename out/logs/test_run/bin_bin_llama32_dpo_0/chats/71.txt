import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Sort bins in descending order of capacity
    bins = np.sort(bins)[::-1]
    
    # Initialize array to store priority scores
    priority_scores = np.zeros_like(bins)
    
    # Perform binary search among all bins
    for i, capacity in enumerate(bins):
        # Calculate the maximum possible amount of item that can be added to this bin
        max_item = capacity - item
        # If max_item is greater than 0, assign it to this bin
        if max_item > 0:
            # Assign the item to this bin and set the priority score
            priority_scores[i] = -max_item
            break
        # If max_item is less than 0, move to the next bin
        else:
            continue
    
    return priority_scores
