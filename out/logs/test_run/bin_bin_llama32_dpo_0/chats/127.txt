```python
import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Ensure item size is not less than bin capacity to avoid unnecessary operations
    item = max(0, item)
    
    # Calculate the difference between the item size and current bin capacity
    # This will give us the priority score of adding the item to the current bin
    priority_score = bins - item
    
    # Calculate the negative of the priority score to get the actual priority
    # This is because the item size minus the bin capacity is equivalent to the priority score
    # Subtracting the negative gives us a value between 0 (no priority) and 1 (maximum priority)
    return -(priority_score)
```
