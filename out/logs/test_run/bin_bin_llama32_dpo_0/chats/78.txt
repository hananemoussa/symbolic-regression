def priority(x, bins):
    """
    Returns a score representing the priority with which to place item in each bin.

    Args:
        x (float): The size of the item.
        bins (np.ndarray): The array of bin capacities.

    Returns:
        np.ndarray: The score array with priority scores for each bin.
    """
    # Calculate the reduction in capacity for each bin
    reduction = np.maximum(bins - x, 0)
    
    # Calculate the relative reduction for each bin
    relative_reduction = reduction / bins
    
    # Compute the scores by negating the relative reductions
    scores = -np.cumsum(relative_reduction)
    
    return scores
