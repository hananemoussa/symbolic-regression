def priority(item_weights: np.ndarray, bins: np.ndarray) -> np.ndarray:

    # Calculate the weighted sum of item weights and bin capacities
    weighted_sum = (item_weights * bins) / (item_weights + bins)

    # Scale the inverse sum to get a priority score
    priority_score = inverse_sum * bins
    
    # Return the priority score array
    return priority_score.astype(np.float64)
