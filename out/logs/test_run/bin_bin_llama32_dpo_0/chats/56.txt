def priority(item: float, bins: np.ndarray) -> np.ndarray:

        bins: Array of capacities for each bin.

    Return:

    """
    # Calculate the signum of the bin capacity minus the item size
    # This introduces non-linear scoring
    signum = (bins - item) / np.linalg.norm(bins) / 2

    # Apply adaptive thresholding to optimize the signum values
    # This adjusts the threshold based on the bin capacity
    threshold = (bins - item) / np.linalg.norm(bins) / 10
    signum[signum > threshold] = 1
    signum[signum < -threshold] = -1

    # Ensure the signum values are within [-1, 1]
    signum = signum * 2 - 1

    return signum
