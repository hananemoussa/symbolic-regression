```python
import numpy as np
import matplotlib.pyplot as plt
from collections import deque

class BinPrioritizer:
    def __init__(self, bins):
        self.bins = bins
        self.priority_queue = deque()

    def add_item(self, item):
        # Find the bin where the item will be added
        for i, bin_capacity in enumerate(self.bins):
            if item <= bin_capacity:
                self.priority_queue.append((item, i))
                break

        # Sort the priority queue
        self.priority_queue.sort(key=lambda x: x[0])

        # Pop the bin with the highest priority
        item, bin_index = self.priority_queue.popleft()

        # If the item is already in the bin, return -inf
        if item == bin_capacity:
            self.priority_queue.append((item, bin_index))
            return -np.inf

        # Return the priority of the bin
        return bin_index

    def plot_priorities(self, bins, items):
        # Plot the item distribution over time
        for item in items:
            bin_index = self.bins.index(item)
            plt.bar(range(len(self.bins)), [self.priority_queue[i][0] for i in range(len(self.bins)) if bin_index == i], width=0.8, label=bin_index)
            plt.xlabel('Bin')
            plt.ylabel('Item')
            plt.title('Item Distribution Over Time')
            plt.legend()
            plt.show()

# Example usage
bins = np.array([100, 200, 300, 400, 500])
items = np.random.randint(1, 501, size=100)

prioritizer = BinPrioritizer(bins)
prioritizer.plot_priorities(bins, items)
prioritizer.add_item(200)
prioritizer.plot_priorities(bins, items)

print(prioritizer.priority_queue)
```

The function `BinPrioritizer` uses a priority queue to find the bin with the highest priority. It adds items to the bin in order of decreasing priority. Once the item is added to the bin, it is removed from the priority queue. The function `plot_priorities` plots the item distribution over time.
