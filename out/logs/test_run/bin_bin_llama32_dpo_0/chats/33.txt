import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """

        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Sort bins by their current capacity in descending order
    bins = np.sort(bins)[::-1]
    
    # Initialize the priority array with zeros
    priority = np.zeros_like(bins)
    
    # Iterate through the sorted bins
    for i, cap in enumerate(bins):
        # Calculate the highest available capacity in the current bin
        available_cap = min(cap, item - bins[i])
        
        # Assign the highest available capacity to the current bin
        priority[i] = available_cap
        
        # Assign the rest of the item to the next bin in the sorted list
        priority[i] += item - available_cap
    
    return priority
