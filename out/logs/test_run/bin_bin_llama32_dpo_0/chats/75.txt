def priority(item: float, bins: np.ndarray) -> np.ndarray:

    # Initialize costs array
    costs = np.zeros_like(bins)

    # Create an identity matrix of the same size as bins
    matrix = np.eye(bins.shape[0])

    # Compute the cost matrix using the Hungarian algorithm
    row_ind, col_ind = linear_sum_assignment(matrix)

    costs[row_ind] = bins[col_ind]
