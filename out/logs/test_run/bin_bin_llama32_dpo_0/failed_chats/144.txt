import numpy as np

def priority_bfs(item: float, bins: np.ndarray, distances: np.ndarray) -> np.ndarray:
    """
    Reuses the original function, priority(), with BFS algorithm instead of depth-first search.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.
        distances: Array of distances to each unvisited bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    visited = np.zeros_like(bins)
    queue = [(item, 0)]  # Initialize the queue with the item and its distance

    while queue:
        distance, index = queue.pop(0)
        if visited[index] == 1:
            continue
        visited[index] = 1
        priority = -(distance - bins[index])
        distance += priority
        distances[index] = distance

        # Explore neighboring bins
        for j in range(len(bins)):
            if bins[j] >= item and not visited[j]:
                queue.append((item + bins[j], distance + priority))

    return distances

def priority_bfs_heappush(item: float, bins: np.ndarray, distances: np.ndarray) -> np.ndarray:
    """
    Reuses the original function, priority(), with priority queue (implemented with heapq) instead of BFS.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.
        distances: Array of distances to each unvisited bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    visited = np.zeros_like(bins)
    heap = [(item, 0)]  # Initialize the heap with the item and its distance

    while heap:
        distance, index = heapq.heappop(heap)
        if visited[index] == 1:
            continue
        visited[index] = 1
        priority = -(distance - bins[index])
        distance += priority
        distances[index] = distance

        # Explore neighboring bins
        for j in range(len(bins)):
            if bins[j] >= item and not visited[j]:
                heapq.heappush(heap, (item + bins[j], distance + priority))

    return distances
