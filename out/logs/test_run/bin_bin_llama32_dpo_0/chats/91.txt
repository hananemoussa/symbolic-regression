import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns a priority score with respect to the capacity of each bin.

    Args:
        item (float): Size of item to be added to the bin.
        bins (np.ndarray): Array of capacities for each bin.

    Return:
        np.ndarray: Array of priority scores for each bin.
    """
    # Compute the total capacity of all bins
    total_capacity = np.sum(bins)

    # Initialize the priority scores with negative infinity
    priority_scores = -np.ones_like(bins)

    # Iterate over each bin
    for i, capacity in enumerate(bins):
        # Calculate the difference between the capacity of the current bin and the item to be added
        bin_diff = capacity - item

        # If the item exceeds the capacity of the current bin, calculate the priority as 1/(total_capacity - bin_diff)
        if bin_diff > 0:
            priority_scores[i] = 1 / (total_capacity - bin_diff)

    return priority_scores
