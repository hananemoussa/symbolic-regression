```python
import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin, optimized to achieve a better score.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Calculate the cumulative sum of bins to get the total capacity
    cumulative_sum = np.cumsum(bins)

    # Calculate the maximum remaining capacity
    max_remaining = bins[-1] - item

    # Calculate the priority for each bin by finding the bin with the most remaining capacity
    priority = cumulative_sum[:-1] - np.maximum(0, cumulative_sum[1:]) + max_remaining

    return priority
```
