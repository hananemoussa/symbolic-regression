import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin using the Greedy Algorithm.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Sort the bins in descending order of capacity
    bins = np.sort(bins)[::-1]
    
    # Initialize the result array
    result = np.zeros_like(bins, dtype=int)
    
    # Assign items to bins using the Greedy Algorithm
    for i, bin_capacity in enumerate(bins):
        # Calculate the priority for the current bin
        priority = -(bin_capacity - item)
        
        # Assign the item to the current bin with the highest priority
        result[i] = i  # Assign to bin with the highest priority
        
        # Update the remaining capacity in the current bin
        bins[i] -= item
    
    return result
