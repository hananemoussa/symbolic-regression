import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns a priority vector indicating the order of bins for each item.

    Args:
        item: The size of the item to be added to the bin.
        bins: An array of capacities for each bin.

    Returns:
        A 1D array of the same size as bins, with priority scores indicating the order of bins.
    """
    # Calculate the ratio of item to bin capacity
    ratios = bins / item
    # Add a small value to each ratio to shift the distribution
    ratios += 1e-10

    # Normalize the ratios to obtain probabilities
    probabilities = ratios / (1 + ratios)

    # Create a priority vector by adding the probabilities of each bin
    return probabilities
