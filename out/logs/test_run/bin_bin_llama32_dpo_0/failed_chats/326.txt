import numpy as np

def binary_search(bin_range: np.ndarray) -> np.ndarray:
    """Performs binary search on the given range to find the optimal bin index."""
    low, high = 0, len(bin_range) - 1
    while low < high:
        mid = (low + high) // 2
        if bin_range[mid] == mid:
            return mid
        elif bin_range[mid] < mid:
            low = mid + 1
        else:
            high = mid - 1
    return low

def priority(item: float, bin_range: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin using binary search."""
    # Sort the bin range in descending order of capacity
    bin_range.sort(reverse=True)

    # Perform binary search to find the optimal bin index
    bin_idx = binary_search(bin_range)

    # Calculate priority score using bin_idx
    priority_score = -item + bin_range[bin_idx]

    return priority_score
