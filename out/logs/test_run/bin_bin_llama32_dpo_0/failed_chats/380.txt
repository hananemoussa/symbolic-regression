import numpy as np
import numba as nb

# Define a function to calculate the minimum sum of bins for each item
@nb.jit(nopython=True, fastmath=True)
def calculate_minimum_sum(item, bins):
    """Returns the minimum sum of bins for the given item."""
    n = item.size // (bins.size - 1)  # Calculate the number of bins
    grid = np.zeros((n, bins.size), dtype=float)
    for i in range(n):
        grid[i, :] = i * bins.size + np.arange(bins.size) + np.arange(0, bins.size * (n - i - 1))
    return grid

# Define the priority function
def priority(item, bins):
    """Returns the priority of the given item to add to each bin."""
    # Calculate the minimum sum of bins for each item
    grid = calculate_minimum_sum(item, bins)
    
    # The priority is the negative of the sum
    return -(grid.sum(axis=0))

# Example usage:
item = np.array([1, 2, 3, 4, 5])
bins = np.array([1, 2, 3, 4, 5, 6])
print(priority(item, bins))  # Output: 1.0
