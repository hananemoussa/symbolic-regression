import numpy as np

def priority(item: float, bins: np.ndarray, num_bins: int = 5) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.
    
    Args:
    item: Size of item to be added to the bin.
    bins: Array of capacities for each bin.
    num_bins: Number of bins for which to calculate the priority score.

    Returns:
    Array of same size as bins with priority score of each bin.
    """
    
    # Calculate the probability distribution of each bin
    probabilities = bins / bins.sum()
    
    # Calculate the cumulative distribution function (CDF)
    cdf = np.cumsum(probabilities)
    
    # Calculate the cumulative sum of the CDF
    cumulative_sum = np.cumsum(cdf)
    
    # Calculate the priority score for each bin
    priority_scores = -cumulative_sum
    
    return priority_scores
