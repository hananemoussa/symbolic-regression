import numpy as np

def priority(item: float, bins: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.
    This function uses heap accumulation to efficiently calculate priority.

    Args:
        item: Size of item to be added to the bin.
        bins: Array of capacities for each bin.

    Return:
        Array of same size as bins with priority score of each bin.
    """
    # Create a copy of the bins array and add the current item to the sum of all previous items
    heap = np.concatenate((bins, np.array([0]*len(bins))), axis=0)
    heap = np.sort(heap)

    # Initialize the priority array
    priority = np.zeros_like(heap)

    # Calculate the priority for each bin
    for i in range(len(heap)):
        # If the current item is less than the sum of all previous items, add the current item to the sum
        if item <= heap[i]:
            heap[i] = item + heap[i]
            # Update the priority array
            priority[i] = -heap[i]

    return priority
```
